function DNAStrand(dna) {
  const mass = dna.toUpperCase().split("");

  const res = mass.map((char) => {
    switch (char) {
      case "A":
        return "T";
      case "T":
        return "A";
      case "C":
        return "G";
      case "G":
        return "C";
    }
  });
  return res.reduce((acc, current) => (acc += current), "");
}
